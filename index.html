Bash (Bourne-again Shell)

Shell = Es un interprete que usa comandos y ordenes para interactuar con el sistema operativo

Comandos de linux basicos

ls = En lista los archivos que se encuentran en el directorio actual
cd = Cambia de directorio
pwd = Imprime el directorio actual de trabajo
clear = Limpia la pantalla
mkdir = Crea un directorio
rmdir = Elimina directorios
touch = Crea archivos
rm = Elimina archivos
mv = Mueve archvios/directorios
cp = Copia archivos
cat = Muestra el contenido de una archivo
chmod = Cambia los permisos de una archivo
grep = busca patrones de texto en archivos

Shebang = Indica la ruta de tu shell predeterminada

Operaciones aritmeticas :

num1="20"
num2="10"
echo $((num1 + num2)) = 30
echo $((num1 - num2)) = 10
echo $((num1 * num2)) = 200
echo $((num1 / num2)) = 2

Operaciones de comparacion

num1="20"

echo $((num1 == num2)) = 0
echo $((num1 != num2)) = 1
echo $((num1 > num2)) = 1
echo $((num1 < num2)) = 0
echo $((num1 >= num2)) = 1
echo $((num1 <= num2)) = 0

-eq = Igual
-ne = No igual
-lt = Menor que
-gt = Mayor que
-ge = Mayor o igual que
-le = Menor o igual que

Matriz = Es una coleccion de elementos

files=("f1.txt" "f2.txt" "f3.txt")

echo ${files[0]} = Selecciona el elemento #
echo ${#files} = Indica el numero de caracteres totales
echo ${files[*]} = Imprime toos lo elementos de la matriz
echo ${#files[@]} = Imprime el numero total de elementos
echo ${files[0]}="f#.txt" = Tambien puedes actualizar cualquier elemento
files+=("f4.txt") = Agrega elementos a la matriz
unset files[#] = Eliminar elemento de la matriz
unset files = Elimina por completo todos los elementos de la matriz
